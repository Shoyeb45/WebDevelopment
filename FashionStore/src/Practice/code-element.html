Program to take n numbers in input and display the sum: 
            #inlcude<bits/stdc++.h>
            using namespace std;

            int main() {
                int n;
                cin >> n;
                vector<int> arr;
                int sum = 0;
                for(int i = 0; i < n; i++) {
                    sum += arr[i];
                }
            }

            cout << sum << "\n";
